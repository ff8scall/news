### 📂 4.2 ai_news_editor.py (JSON 파싱 방어 & Fallback 로직)

```python
import json, os, re, time, logging
from ai_writer import AIWriter

class NewsEditor:
    def __init__(self, writer=None): self.writer = writer if writer else AIWriter()

    def _score_articles(self, articles):
        """[V0.2.1 Hotfix] AI 답변이 불완전할 경우를 대비한 숫자 추출 Fallback 탑재"""
        prompt = "[TASK]: Score these news [1-10]\n" + "\n".join([f"- {a['title']}" for a in articles])
        res = self.writer.generate_content(prompt, role="processing")
        scores = []
        try:
            match = re.search(r'\[(.*?)\]', res, re.DOTALL)
            if match: scores = json.loads(f"[{match.group(1)}]")
            else: scores = [int(n) for n in re.findall(r'\d+', res)] # [V0.2.1 Fix] Fallback to regex numbers
        except: scores = [7] * len(articles)
        
        final = []
        for i, article in enumerate(articles):
            score = scores[i] if i < len(scores) else 7
            if score >= 7: final.append(article)
        return final if final else [articles[0]]
```
